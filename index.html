x = [i for i in range(1, 10)]
y = "X"
def printTable():
    for i in range(0, 9, 3):
        for j in range(3):
            print(x[j + i], end="|")
        print("\n" + "-" * 7)
printTable()
for i in range(9):
    x[int(input(f"Enter your move ({y}) :")) - 1] = y
    if x[0] == x[1] == x[2] or x[0] == x[3] == x[6] or x[0] == x[4] == x[8] or x[2] == x[1] == x[0] or x[2] == x[5] == x[8] or x[2] == x[4] == x[6] or x[3] == x[4] == x[5] or x[1] == x[4] == x[7]:
        print(f"{y} Win")
        printTable()
        break
    if i == 8:
        print("Draw Game!")
    printTable()
    if y == "X":
        y = "O"
    else:
        y = "X"
